01. modal popup - 기본

알림

모달 팝업 내용입니다.

<button type="button" class="form-button form-button-secondary js-modal-open" data-modal-target="#guide-modal">모달 열기</button>

<!-- modal popup 기본 예제 -->
<div id="guide-modal" class="ui-modal" role="dialog" aria-modal="true" aria-labelledby="guide-modal-title" aria-hidden="true">
    <div class="ui-modal-dimmed js-modal-close"></div>
    <div class="ui-modal-dialog">
        <div class="ui-modal-header"><h3 id="guide-modal-title" class="ui-modal-title">알림</h3></div>

// modal popup 동작
const modal = document.querySelector('#guide-modal');
const openButton = document.querySelector('[data-modal-target="#guide-modal"]');
const closeButtons = modal.querySelectorAll('.js-modal-close');

openButton.addEventListener('click', function () {
    modal.classList.add('is-open');
    modal.setAttribute('aria-hidden', 'false');
    modal.querySelector('button.js-modal-close').focus();
});

closeButtons.forEach(function (button) {
    button.addEventListener('click', function () {
        modal.classList.remove('is-open');
        modal.setAttribute('aria-hidden', 'true');
        openButton.focus();
    });
});
  • 모달·알럿·컨펌은 PC에서 최대 너비 400px를 사용하고, 작은 화면에서는 가용 너비에 맞게 축소한다.
  • 열기 버튼의 data-modal-target과 모달 id를 연결한다.
  • 열림 시 초점을 내부로 이동하고 Tab 포커스를 레이어 안에서 순환시킨다.
  • Esc 또는 닫기 버튼으로 닫은 뒤 실행 버튼으로 포커스를 복원한다.

  • 공통 이미지 레이어의 라운드, 배경과 구분선은 프로젝트 디자인에 맞게 적용한다.
  • 이미지와 하단 조작 영역은 경계를 구분하되 색상과 선의 형태는 프로젝트 기준을 따른다.
  • 배너 이미지가 정보를 포함하면 목적과 내용을 alt로 제공하고, 장식 이미지는 alt를 비운다.

  • 타이틀형은 하단 확인 유형과 상단 닫기·하단 확인 유형을 목적에 맞게 구분한다.
  • 하단 버튼이 없는 유형은 상단 닫기 버튼에 구체적인 aria-label을 제공하고 닫은 뒤 열기 버튼으로 포커스를 복원한다.
  • 슬라이드 반복하지 않고 첫 번째·마지막 이미지에서 이동할 수 없는 방향 버튼을 노출하지 않는다.
  • 페이지네이션 유형은 번호 또는 도트 중 목적에 맞는 한 가지 유형을 사용한다.
  • 숨겨진 탭이나 팝업 안에서 Swiper를 초기화할 경우 노출 시점에 update() 를 실행해 크기와 위치를 다시 계산한다.
  • 오늘 하루 안보기 확인 버튼과 동일한 너비로 배치하고 두 항목을 시각적으로 구분한다.
  • 선택 상태를 저장하는 프로젝트 로직과 연결하고 확인 버튼과 명확히 구분한다.

// bottom sheet 동작
const sheet = document.querySelector('#guide-bottom-sheet');
const sheetOpenButton = document.querySelector('[data-bottom-sheet-target="#guide-bottom-sheet"]');

sheetOpenButton.addEventListener('click', function () {
    sheet.classList.add('is-opening');
    sheet.classList.add('is-open');
    sheet.setAttribute('aria-hidden', 'false');
    sheet.querySelector('button.js-bottom-sheet-close').focus();

    window.requestAnimationFrame(function () {
        window.requestAnimationFrame(function () {
            sheet.classList.remove('is-opening');
        });
    });
});

sheet.querySelectorAll('.js-bottom-sheet-close').forEach(function (button) {
    button.addEventListener('click', function () {
        sheet.classList.add('is-closing');
        sheet.setAttribute('aria-hidden', 'true');

        window.setTimeout(function () {
            sheet.classList.remove('is-open', 'is-closing');
            sheetOpenButton.focus();
        }, 250);
    });
});
  • 열 때는 패널이 아래에서 올라오며 딤이 나타나고, 닫을 때는 반대로 전환되는 0.25초 모션을 사용한다.
  • 동작 줄이기 설정에서는 열림·닫힘 모션을 즉시 완료한다.
  • 바텀시트 콘텐츠 영역은 기본 최소 높이 120px를 사용하고, 내용이 많아지면 자연스럽게 확장되도록 한다.
  • 바텀시트는 PC에서 최대 480px로 하단 중앙에 배치하고, 480px 이하에서는 화면 너비에 맞게 사용한다.
  • 핸들은 바텀시트임을 알리는 장식 요소로 사용하며, 단독으로 닫기 기능을 부여하지 않는다.
  • 드래그로 닫는 기능은 보조 수단으로만 제공하고, 닫기 기능이 필요하면 접근 가능한 이름을 가진 button과 열기 버튼으로의 포커스 복원을 함께 구현한다.
  • 바텀시트는 화면 하단에서 제공하는 선택·보조 작업에 사용하고 열린 동안 Tab 포커스를 내부에서 순환시킨다.
  • 버튼은 가용 너비를 동일하게 나누고, 480px 이하에서는 세로로 전환해 텍스트 잘림을 방지한다.
  • 버튼 수와 문구 길이를 변경한 뒤 모바일 최소 해상도에서 겹침과 가로 스크롤을 확인한다.

01. toast popup - 기본

설정이 저장되었습니다. 
<button type="button" class="form-button form-button-secondary js-toast-open" data-toast-target="#guide-toast">토스트 열기</button>

<!-- toast popup 기본 예제 -->
<div id="guide-toast" class="ui-toast" role="status" aria-live="polite">설정이 저장되었습니다.</div>
// toast popup 동작
const toastButton = document.querySelector('[data-toast-target="#guide-toast"]');
const toast = document.querySelector('#guide-toast');
let toastTimer = null;

toastButton.addEventListener('click', function () {
    window.clearTimeout(toastTimer);
    toast.classList.add('is-open');
    toastTimer = window.setTimeout(function () {
        toast.classList.remove('is-open');
    }, 3000);
});
토스트는 작업 결과처럼 짧은 상태 안내에 사용하며, 중요한 확인이나 사용자 선택이 필요한 경우 모달을 사용한다.
